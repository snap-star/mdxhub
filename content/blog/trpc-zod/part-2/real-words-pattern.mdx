### 5.1 Building a Context with Authentication

**File: `src/server/context.ts`**

```typescript
import { CreateHTTPContextOptions } from '@trpc/server/adapters/standalone';

interface User {
  id: string;
  email: string;
  role: 'user' | 'admin';
}

export async function createContext({ req }: CreateHTTPContextOptions) {
  // Extract token from header
  const authHeader = req.headers.authorization;
  
  if (!authHeader?.startsWith('Bearer ')) {
    return { user: null }; // Anonymous user
  }
  
  const token = authHeader.slice(7);
  
  try {
    // Verify JWT and get user
    const payload = await verifyToken(token);
    const user = await db.user.findUnique({ where: { id: payload.sub } });
    
    return { user }; // Authenticated context
  } catch {
    return { user: null }; // Invalid token
  }
}

export type Context = Awaited<ReturnType<typeof createContext>>;
```

**File: `src/server/trpc.ts`** — Updated with context

```typescript
import { initTRPC, TRPCError } from '@trpc/server';
import type { Context } from './context';

const t = initTRPC.context<Context>().create();

export const router = t.router;
export const publicProcedure = t.procedure;

// Middleware: require authentication
const enforceAuth = t.middleware(({ ctx, next }) => {
  if (!ctx.user) {
    throw new TRPCError({ code: 'UNAUTHORIZED' });
  }
  return next({
    ctx: {
      ...ctx,
      user: ctx.user, // Now guaranteed non-null
    },
  });
});

// Middleware: require admin role
const enforceAdmin = t.middleware(({ ctx, next }) => {
  if (!ctx.user) {
    throw new TRPCError({ code: 'UNAUTHORIZED' });
  }
  if (ctx.user.role !== 'admin') {
    throw new TRPCError({ code: 'FORBIDDEN', message: 'Admin access required' });
  }
  return next({ ctx });
});

// Create protected procedures
export const authedProcedure = t.procedure.use(enforceAuth);
export const adminProcedure = t.procedure.use(enforceAdmin);
```

### 5.3 React Client with Auth Integration

```typescript
// src/client/trpc.ts — with auth token management
import { createTRPCReact } from '@trpc/react-query';
import { httpBatchLink, retryLink } from '@trpc/client';
import type { AppRouter } from '../server/routers';

export const trpc = createTRPCReact<AppRouter>();

export function TrpcProvider({ children }: { children: React.ReactNode }) {
  const [queryClient] = useState(() => new QueryClient({
    defaultOptions: {
      queries: {
        // Global error handling
        meta: {
          onError: (error: TRPCClientError<AppRouter>) => {
            if (error.data?.code === 'UNAUTHORIZED') {
              window.location.href = '/login';
            }
          },
        },
      },
    },
  }));

  const trpcClient = trpc.createClient({
    links: [
      // Retry failed requests intelligently
      retryLink({
        retry: (opts) => {
          // Don't retry mutations
          if (opts.op.type === 'mutation') return false;
          // Don't retry auth errors
          if (opts.error.data?.code === 'UNAUTHORIZED') return false;
          // Default retry logic
          return opts.attempts < 3;
        },
      }),
      httpBatchLink({
        url: '/api/trpc',
        headers() {
          const token = localStorage.getItem('token');
          return token ? { Authorization: `Bearer ${token}` } : {};
        },
      }),
    ],
  });

  return (
    <trpc.Provider client={trpcClient} queryClient={queryClient}>
      <QueryClientProvider client={queryClient}>
        {children}
      </QueryClientProvider>
    </trpc.Provider>
  );
}
```

```typescript
import { z } from 'zod';

// Reusable field validators
export const schemas = {
  // IDs
  id: z.string().uuid(),
  cuid: z.string().cuid2(),
  
  // User fields
  email: z.string().email().toLowerCase().trim(),
  password: z.string().min(8).max(100)
    .regex(/[A-Z]/, 'Must contain uppercase')
    .regex(/[a-z]/, 'Must contain lowercase')
    .regex(/[0-9]/, 'Must contain number'),
  
  username: z.string().min(3).max(30)
    .regex(/^[a-zA-Z0-9_]+$/, 'Only letters, numbers, underscores'),
  
  // Content
  title: z.string().min(1).max(200).trim(),
  description: z.string().max(2000).trim().optional(),
  
  // Pagination
  page: z.coerce.number().int().min(1).default(1),
  limit: z.coerce.number().int().min(1).max(100).default(20),
  
  // Sorting
  sortOrder: z.enum(['asc', 'desc']).default('desc'),
};

// Reusable pagination input
export const paginationInput = z.object({
  page: schemas.page,
  limit: schemas.limit,
  search: z.string().optional(),
  sortBy: z.string().optional(),
  sortOrder: schemas.sortOrder,
});

export type PaginationInput = z.infer<typeof paginationInput>;

// Usage in routers
export const postRouter = router({
  list: publicProcedure
    .input(paginationInput)
    .query(async ({ input }) => {
      const { page, limit } = input;
      const skip = (page - 1) * limit;
      
      const [posts, total] = await Promise.all([
        db.post.findMany({ skip, take: limit }),
        db.post.count(),
      ]);
      
      return {
        data: posts,
        meta: {
          page,
          limit,
          total,
          totalPages: Math.ceil(total / limit),
        },
      };
    }),
});
```

### 6.2 Schema Composition & Refinement

```typescript
import { z } from 'zod';

// Base schemas that can be extended
const timestampFields = {
  createdAt: z.date().optional(),
  updatedAt: z.date().optional(),
};

// Base entity schema
const entitySchema = z.object({
  id: z.string().uuid(),
  ...timestampFields,
});

// Extending schemas
type Entity = z.infer<typeof entitySchema>;

const userBaseSchema = z.object({
  name: z.string().min(1),
  email: z.string().email(),
});

// Create User Input (no ID needed)
const createUserSchema = userBaseSchema.extend({
  password: z.string().min(8),
  role: z.enum(['user', 'admin']).default('user'),
});

// Update User Input (all fields optional)
const updateUserSchema = userBaseSchema.partial();

// Full User (from database)
const userSchema = entitySchema.merge(userBaseSchema).extend({
  role: z.enum(['user', 'admin']),
  avatar: z.string().url().nullable(),
});

// Advanced refinement — custom validation
const passwordSchema = z.object({
  password: z.string().min(8),
  confirmPassword: z.string(),
}).refine(
  (data) => data.password === data.confirmPassword,
  {
    message: "Passwords don't match",
    path: ['confirmPassword'], // Error will be attached to this field
  }
);

// Transform — modify data after validation
const slugSchema = z.string()
  .min(1)
  .transform((val) => val.toLowerCase().replace(/\s+/g, '-'));

// Async refinement — check against database
const uniqueEmailSchema = z.string().email()
  .refine(
    async (email) => {
      const existing = await db.user.findUnique({ where: { email } });
      return existing === null;
    },
    { message: 'Email already in use' }
  );
```

```typescript
// Define output schemas for API contracts
const userOutputSchema = z.object({
  id: z.string().uuid(),

  // Explicitly exclude password from output!

const postOutputSchema = z.object({

});

    .query(async ({ input }) => {
      const post = await db.post.findUnique({
        where: input,
        include: { author: true },
      });
      
      if (!post) throw new TRPCError({ code: 'NOT_FOUND' });
      
      // Output is automatically validated against postOutputSchema
      // If the shape doesn't match, tRPC throws an error
      return post;
    }),
});
```

### 7.1 Creating Reusable Middleware

```typescript
import { initTRPC, TRPCError } from '@trpc/server';
import type { Context } from './context';

const t = initTRPC.context<Context>().create();

// Logging middleware
const logger = t.middleware(async ({ path, type, rawInput, next }) => {
  const start = Date.now();
  
  console.log(`→ ${type.toUpperCase()} ${path}`);
  
  const result = await next();
  
  const duration = Date.now() - start;
  console.log(`← ${type.toUpperCase()} ${path} — ${result.ok ? 'OK' : 'ERR'} (${duration}ms)`);
  
  return result;
});

// Rate limiting middleware
const rateLimit = t.middleware(async ({ ctx, path, next }) => {
  const key = `${ctx.user?.id ?? ctx.ip}:${path}`;
  const limit = await checkRateLimit(key);
  
  if (limit.exceeded) {
    throw new TRPCError({
      code: 'TOO_MANY_REQUESTS',
      message: `Rate limit exceeded. Retry after ${limit.retryAfter}s`,
    });
  }
  
  return next();
});

// Input sanitization middleware
const sanitize = t.middleware(async ({ rawInput, next }) => {
  if (typeof rawInput === 'object' && rawInput !== null) {
    // Strip HTML tags from string inputs
    const sanitized = Object.fromEntries(
      Object.entries(rawInput).map(([key, value]) => [
        key,
        typeof value === 'string' ? stripHtml(value) : value,
      ])
    );
    return next({ rawInput: sanitized });
  }
  return next();
});

// Compose middleware into procedures
export const loggedProcedure = t.procedure.use(logger);
export const rateLimitedProcedure = t.procedure.use(logger).use(rateLimit);
export const sanitizedProcedure = t.procedure.use(logger).use(sanitize);

// Usage
export const apiRouter = router({
  // All calls are logged
  health: loggedProcedure.query(() => ({ status: 'ok' })),
  
  // Rate limited + logged
  search: rateLimitedProcedure
    .input(z.object({ q: z.string().min(1) }))
    .query(async ({ input }) => {
      return performSearch(input.q);
    }),
  
  // Sanitized input + logged
  createPost: sanitizedProcedure
    .input(z.object({ title: z.string(), content: z.string() }))
    .mutation(async ({ input }) => {
      // input.title and input.content are now HTML-safe
      return db.post.create({ data: input });
    }),
});
```

### 7.2 Context Enrichment Pattern

```typescript
import { t } from './trpc-init';

// Middleware that enriches context with additional data
const withPost = t.middleware(async ({ ctx, rawInput, next }) => {
  const input = rawInput as { postId: string };
  
  const post = await db.post.findUnique({
    where: { id: input.postId },
  });
  
  if (!post) {
    throw new TRPCError({ code: 'NOT_FOUND', message: 'Post not found' });
  }
  
  // Pass post to context so procedures don't need to re-fetch
  return next({ ctx: { ...ctx, post } });
});

// Middleware that enriches with user's organization
const withOrganization = t.middleware(async ({ ctx, next }) => {
  if (!ctx.user) throw new TRPCError({ code: 'UNAUTHORIZED' });
  
  const org = await db.organization.findFirst({
    where: { members: { some: { userId: ctx.user.id } } },
  });
  
  if (!org) {
    throw new TRPCError({ code: 'FORBIDDEN', message: 'No organization found' });
  }
  
  return next({ ctx: { ...ctx, organization: org } });
});

// Use enriched context
export const postRouter = router({
  // post is pre-loaded by middleware
  getDetails: authedProcedure
    .input(z.object({ postId: z.string().uuid() }))
    .use(withPost)
    .query(async ({ ctx }) => {
      // ctx.post is available and typed!
      return {
        ...ctx.post,
        canEdit: ctx.post.authorId === ctx.user.id,
      };
    }),

  // organization is pre-loaded
  getOrgPosts: authedProcedure
    .use(withOrganization)
    .query(async ({ ctx }) => {
      // ctx.organization is available!
      return db.post.findMany({
        where: { orgId: ctx.organization.id },
      });
    }),
});
```

```typescript
// Server: Subscription router
import { observable } from '@trpc/server/observable';
import { EventEmitter } from 'events';

const messageEmitter = new EventEmitter();

export const messageRouter = router({
  onNewMessage: authedProcedure
    .input(z.object({ roomId: z.string().uuid() }))
    .subscription(({ input }) => {
      // Return an observable that emits new messages
      return observable<{ id: string; content: string; sender: string }>((emit) => {
        const onMessage = (msg: any) => {
          if (msg.roomId === input.roomId) {
            emit.next(msg);
          }
        };
        
        messageEmitter.on('message', onMessage);
        
        // Cleanup when unsubscribed
        return () => {
          messageEmitter.off('message', onMessage);
        };
      });
    }),

  send: authedProcedure
    .input(z.object({
      roomId: z.string().uuid(),
      content: z.string().min(1).max(2000),
    }))
    .mutation(async ({ ctx, input }) => {
      const message = await db.message.create({
        data: {
          ...input,
          senderId: ctx.user.id,
        },
      });
      
      // Notify subscribers
      messageEmitter.emit('message', message);
      
      return message;
    }),
});
```

### 8.2 React Client for Subscriptions

```typescript
'use client';

import { trpc } from './trpc';

export function ChatRoom({ roomId }: { roomId: string }) {
  const [messages, setMessages] = useState<Message[]>([]);

  // Subscribe to real-time messages
  trpc.message.onNewMessage.useSubscription(
    { roomId },
    {
      onData: (newMessage) => {
        setMessages((prev) => [...prev, newMessage]);
      },
      onError: (err) => {
        console.error('Subscription error:', err);
      },
    }
  );

  const sendMessage = trpc.message.send.useMutation();

  return (
    <div>
      <div className="messages">
        {messages.map((msg) => (
          <div key={msg.id}>
            <strong>{msg.sender}:</strong> {msg.content}
          </div>
        ))}
      </div>
      
      <form
        onSubmit={async (e) => {
          e.preventDefault();
          const input = e.currentTarget.elements.namedItem('msg') as HTMLInputElement;
          await sendMessage.mutateAsync({ roomId, content: input.value });
          input.value = '';
        }}
      >
        <input name="msg" maxLength={2000} />
        <button type="submit">Send</button>
      </form>
    </div>
  );
}
```

```typescript
// app/actions.ts
'use server';

import { createCaller } from '@/server/routers';
import { revalidatePath } from 'next/cache';
import { z } from 'zod';

const createTodoSchema = z.object({
  text: z.string().min(1),
});

export async function createTodoAction(formData: FormData) {
  const parsed = createTodoSchema.safeParse({
    text: formData.get('text'),
  });
  
  if (!parsed.success) {
    return { error: parsed.error.flatten().fieldErrors };
  }
  
  const caller = createCaller({ user: null });
  await caller.todo.create(parsed.data);
  
  revalidatePath('/');
  return { success: true };
}
```

### 10.1 Cursor-Based Pagination (Recommended)

```typescript
// Server: Cursor pagination implementation
export const postRouter = router({
  infiniteList: publicProcedure
    .input(
      z.object({
        cursor: z.string().optional(), // Last item's ID
        limit: z.number().min(1).max(50).default(10),
        category: z.string().optional(),
      })
    )
    .query(async ({ input }) => {
      const { cursor, limit, category } = input;

      const posts = await db.post.findMany({
        take: limit + 1, // Take one extra to determine if there's more
        cursor: cursor ? { id: cursor } : undefined,
        where: category ? { category } : undefined,
        orderBy: { createdAt: 'desc' },
      });

      let nextCursor: typeof cursor = undefined;

      // If we got more items than requested, there's a next page
      if (posts.length > limit) {
        const nextItem = posts.pop(); // Remove the extra item
        nextCursor = nextItem!.id;
      }

      return {
        items: posts,
        nextCursor,
      };
    }),
});
```
