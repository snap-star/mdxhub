### 11.2 Factory Pattern for CRUD Routers

```typescript
// src/server/lib/crud-router.ts — Generic CRUD router factory
import { z } from 'zod';
import { router, publicProcedure, authedProcedure } from '../trpc';

interface CrudOptions<TCreate, TUpdate, TOutput> {
  name: string;
  createSchema: z.ZodType<TCreate>;
  updateSchema: z.ZodType<TUpdate>;
  outputSchema: z.ZodType<TOutput>;
  db: {
    create: (data: TCreate) => Promise<TOutput>;
    findMany: (params: { skip?: number; take?: number }) => Promise<TOutput[]>;
    findById: (id: string) => Promise<TOutput | null>;
    update: (id: string, data: Partial<TUpdate>) => Promise<TOutput>;
    delete: (id: string) => Promise<void>;
    count: () => Promise<number>;
  };
}

export function createCrudRouter<TCreate, TUpdate, TOutput>(
  options: CrudOptions<TCreate, TUpdate, TOutput>
) {
  const { createSchema, updateSchema, db } = options;

  return router({
    list: publicProcedure
      .input(paginationInput)
      .query(async ({ input }) => {
        const skip = (input.page - 1) * input.limit;
        const [items, total] = await Promise.all([
          db.findMany({ skip, take: input.limit }),
          db.count(),
        ]);
        return { items, meta: { page: input.page, limit: input.limit, total } };
      }),

    getById: publicProcedure
      .input(z.object({ id: z.string().uuid() }))
      .query(async ({ input }) => {
        const item = await db.findById(input.id);
        if (!item) throw new TRPCError({ code: 'NOT_FOUND' });
        return item;
      }),

    create: authedProcedure
      .input(createSchema)
      .mutation(async ({ input }) => db.create(input)),

    update: authedProcedure
      .input(z.object({ id: z.string().uuid(), data: updateSchema }))
      .mutation(async ({ input }) => db.update(input.id, input.data)),

    delete: authedProcedure
      .input(z.object({ id: z.string().uuid() }))
      .mutation(async ({ input }) => db.delete(input.id)),
  });
}

// Usage:
const tagRouter = createCrudRouter({
  name: 'tag',
  createSchema: z.object({ name: z.string().min(1) }),
  updateSchema: z.object({ name: z.string().min(1) }).partial(),
  outputSchema: z.object({ id: z.string(), name: z.string() }),
  db: tagDb,
});
```

```typescript
// src/server/lib/data-loader.ts — DataLoader pattern for N+1 prevention
import DataLoader from 'dataloader';

// Batch load users by ID — prevents N+1 queries
const userLoader = new DataLoader<string, User>(async (userIds) => {
  const users = await db.user.findMany({
    where: { id: { in: [...userIds] } },
  });
  
  // Return in same order as keys
  const userMap = new Map(users.map((u) => [u.id, u]));
  return userIds.map((id) => userMap.get(id) ?? null);
});

// Context factory with loaders
export async function createContext() {
  return {
    loaders: {
      user: userLoader,
      // Add more loaders...
    },
  };
}

export const postRouter = router({
```

### 12.2 Response Caching

```typescript
import { redis } from './redis-client';

// Middleware: Redis response caching
const cached = (ttlSeconds: number) => t.middleware(async ({ path, type, next }) => {
  // Only cache queries
  if (type !== 'query') return next();
  
  const cacheKey = `trpc:${path}`;
  const cached = await redis.get(cacheKey);
  
  if (cached) {
    return { result: { data: JSON.parse(cached) } };
  }
  
  const result = await next();
  
  if (result.ok) {
    await redis.setex(cacheKey, ttlSeconds, JSON.stringify(result.data));
  }
  
  return result;
});

// Usage:
export const postRouter = router({
  getById: publicProcedure
    .input(z.object({ id: z.string().uuid() }))
    .use(cached(60)) // Cache for 60 seconds
    .query(async ({ input }) => {
      return db.post.findUnique({ where: input });
    }),

  // Cache expensive computations longer
  getTrending: publicProcedure
    .use(cached(300)) // Cache for 5 minutes
    .query(async () => {
      // Expensive aggregation query...
      return computeTrendingPosts();
```

```typescript
'use client';

import { trpc } from './trpc';

// Optimistic updates with rollback
function useOptimisticToggle() {
  const utils = trpc.useUtils();
  
  return trpc.post.toggleLike.useMutation({
    // Optimistically update UI
    onMutate: async ({ postId }) => {
      await utils.post.getById.cancel({ id: postId });
      
      const previous = utils.post.getById.getData({ id: postId });
      
      utils.post.getById.setData({ id: postId }, (old) => {
        if (!old) return old;
        return {
          ...old,
          liked: !old.liked,
          likeCount: old.liked ? old.likeCount - 1 : old.likeCount + 1,
        };
      });
      
      return { previous };
    },
    
    onError: (_err, { postId }, context) => {
      utils.post.getById.setData({ id: postId }, context?.previous);
    },
    
    onSettled: (_data, _err, { postId }) => {
      utils.post.getById.invalidate({ id: postId });
    },
  });
}

// Stale-while-revalidate pattern
function useRealtimeData() {
  return trpc.dashboard.stats.useQuery(undefined, {
    refetchInterval: 30000,      // Refetch every 30s
    staleTime: 10000,            // Data fresh for 10s
    refetchOnWindowFocus: true,  // Refresh when user returns
    placeholderData: (prev) => prev, // Show old data while loading
  });
}

function usePrefetchPost() {
  const utils = trpc.useUtils();
  
  return {
    prefetchPost: (postId: string) => {
      utils.post.getById.prefetch({ id: postId });
      utils.post.comments.prefetch({ postId });
```

### 13.1 Role-Based Access Control (RBAC)

```typescript
// src/server/auth/permissions.ts
import { TRPCError } from '@trpc/server';

// Define all possible permissions
const Permissions = {
  post: ['create', 'read', 'update', 'delete', 'publish'] as const,
  user: ['read', 'update', 'delete', 'manageRoles'] as const,
  comment: ['create', 'read', 'update', 'delete', 'moderate'] as const,
} as const;

type Permission = {
  [K in keyof typeof Permissions]: `${K}:${(typeof Permissions)[K][number]}`;
}[keyof typeof Permissions];

// Role definitions
const RolePermissions: Record<string, Permission[]> = {
  admin: [
    'post:create', 'post:read', 'post:update', 'post:delete', 'post:publish',
    'user:read', 'user:update', 'user:delete', 'user:manageRoles',
    'comment:create', 'comment:read', 'comment:update', 'comment:delete', 'comment:moderate',
  ],
  editor: [
    'post:create', 'post:read', 'post:update', 'post:publish',
    'user:read',
    'comment:create', 'comment:read', 'comment:moderate',
  ],
  author: [
    'post:create', 'post:read', 'post:update',
    'user:read',
    'comment:create', 'comment:read',
  ],
  user: [
    'post:read',
    'user:read', 'user:update',
    'comment:create', 'comment:read',
  ],
};

// Middleware factory for permission checking
const requirePermission = (permission: Permission) => 
  t.middleware(({ ctx, next }) => {
    if (!ctx.user) throw new TRPCError({ code: 'UNAUTHORIZED' });
    
    const userPermissions = RolePermissions[ctx.user.role] ?? [];
    
    if (!userPermissions.includes(permission)) {
      throw new TRPCError({ 
        code: 'FORBIDDEN', 
        message: `Missing permission: ${permission}` 
      });
    }
    
    return next({ ctx });
  });

// Usage:
export const postRouter = router({
  create: authedProcedure
    .use(requirePermission('post:create'))
    .input(createPostSchema)
    .mutation(async ({ ctx, input }) => {
      // User guaranteed to have post:create permission
      return db.post.create({ data: { ...input, authorId: ctx.user.id } });
    }),

  publish: authedProcedure
    .use(requirePermission('post:publish'))
    .input(z.object({ id: z.string().uuid() }))
    .mutation(async ({ input }) => {
      return db.post.update({
        where: input,
        data: { status: 'published', publishedAt: new Date() },
      });
    }),

  moderate: authedProcedure
    .use(requirePermission('comment:moderate'))
    .input(z.object({ commentId: z.string().uuid(), action: z.enum(['approve', 'reject']) }))
    .mutation(async ({ input }) => { /* ... */ }),
});
```

### 13.2 Multi-Tenant Context

```typescript
// src/server/context/organization.ts
interface OrgContext {
  user: User;
  organization: Organization;
  membership: OrganizationMember;
}

const withOrganization = t.middleware(async ({ ctx, next }) => {
  if (!ctx.user) throw new TRPCError({ code: 'UNAUTHORIZED' });
  
  // Extract org from header or user default
  const orgId = ctx.req.headers['x-organization-id'] as string 
    ?? ctx.user.defaultOrgId;
  
  const [organization, membership] = await Promise.all([
    db.organization.findUnique({ where: { id: orgId } }),
    db.organizationMember.findUnique({
      where: { userId_orgId: { userId: ctx.user.id, orgId } },
    }),
  ]);
  
  if (!organization || !membership) {
    throw new TRPCError({ code: 'FORBIDDEN', message: 'Organization access denied' });
  }
  
  return next({
    ctx: { ...ctx, organization, membership } as OrgContext,
  });
});

// Organization-scoped procedures
const orgProcedure = authedProcedure.use(withOrganization);

// Usage:
export const projectRouter = router({
  list: orgProcedure.query(async ({ ctx }) => {
    // Automatically scoped to user's org!
    return db.project.findMany({
      where: { organizationId: ctx.organization.id },
    });
  }),

  create: orgProcedure
    .input(z.object({ name: z.string().min(1) }))
    .mutation(async ({ ctx, input }) => {
      return db.project.create({
        data: {
          ...input,
          organizationId: ctx.organization.id,
          createdById: ctx.user.id,
        },
      });
    }),
});
```

```typescript
// Gateway: Aggregates multiple tRPC services
import { router } from './trpc';

// Client for User Service
const userService = createTRPCProxyClient<AppRouter>({
  links: [httpBatchLink({ url: 'http://user-service:3001' })],
});

// Client for Order Service
const orderService = createTRPCProxyClient<OrderRouter>({
  links: [httpBatchLink({ url: 'http://order-service:3002' })],
});

// Client for Inventory Service
const inventoryService = createTRPCProxyClient<InventoryRouter>({
  links: [httpBatchLink({ url: 'http://inventory-service:3003' })],
});

// Gateway router — composes all services
export const gatewayRouter = router({
  user: createServiceProxy(userService),
  order: createServiceProxy(orderService),
  inventory: createServiceProxy(inventoryService),
  
  // Cross-service queries
  userOrders: authedProcedure
    .input(z.object({ userId: z.string().uuid() }))
    .query(async ({ input }) => {
      // Parallel calls to multiple services
      const [user, orders] = await Promise.all([
        userService.user.getById.query({ id: input.userId }),
        orderService.order.byUser.query({ userId: input.userId }),
      ]);
      
      return { user, orders };
    }),

  // Saga pattern: distributed transaction
  placeOrder: authedProcedure
    .input(z.object({
      items: z.array(z.object({
        productId: z.string().uuid(),
        quantity: z.number().int().positive(),
      })),
      shippingAddress: z.object({ /* ... */ }),
    }))
    .mutation(async ({ ctx, input }) => {
      // Step 1: Check inventory
      const stockChecks = await Promise.all(
        input.items.map((item) =>
          inventoryService.inventory.check.query({
            productId: item.productId,
            quantity: item.quantity,
          })
        )
      );
      
      if (stockChecks.some((check) => !check.available)) {
        throw new TRPCError({ code: 'PRECONDITION_FAILED', message: 'Out of stock' });
      }
      
      // Step 2: Reserve inventory
      await Promise.all(
        input.items.map((item) =>
          inventoryService.inventory.reserve.mutate({
            productId: item.productId,
            quantity: item.quantity,
          })
        )
      );
      
      // Step 3: Create order
      const order = await orderService.order.create.mutate({
        userId: ctx.user.id,
        items: input.items,
        shippingAddress: input.shippingAddress,
      });
      
      // If any step fails, implement compensation (rollback)
      
      return order;
    }),
});
```
