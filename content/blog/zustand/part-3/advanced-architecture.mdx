## 16. Custom Middleware Development

### Complete Middleware Template

```jsx
// middleware/template.js
/**
 * Custom Zustand Middleware Template
 * 
 * @param {Function} config - Store initializer function
 * @param {Object} options - Middleware options
 * @returns {Function} Enhanced store initializer
 */
const customMiddleware = (config, options = {}) => (set, get, api) => {
  // ─── Setup: Initialize middleware state ───
  const middlewareState = {
    callCount: 0,
    startTime: Date.now(),
  };

  // ─── Wrap set() ───
  const customSet = (...args) => {
    // Before set
    middlewareState.callCount++;
    const prevState = get();

    // Call original set
    set(...args);

    // After set
    const nextState = get();

    // Post-processing
    console.log(`[Middleware] Set call #${middlewareState.callCount}`);

    return nextState;
  };

  // ─── Wrap get() ───
  const customGet = () => {
    // Can add logging, tracking, etc.
    return get();
  };

  // ─── Enhance API ───
  const enhancedApi = {
    ...api,
    // Add custom methods to the store API
    getMiddlewareStats: () => ({
      calls: middlewareState.callCount,
      uptime: Date.now() - middlewareState.startTime,
    }),
  };

  // ─── Create store with wrapped functions ───
  const store = config(customSet, customGet, enhancedApi);

  // ─── Return enhanced store ───
  return {
    ...store,
    // Expose middleware methods on the store
    getMiddlewareStats: enhancedApi.getMiddlewareStats,
  };
};
```

### Comprehensive Logger Middleware

```jsx
// middleware/comprehensiveLogger.js
const comprehensiveLogger = (config, options = {}) => (set, get, api) => {
  const {
    enabled = process.env.NODE_ENV === 'development',
    collapsed = true,
    colors = {
      title: '#03A9F4',
      prev: '#9E9E9E',
      action: '#03A9F4',
      next: '#4CAF50',
      error: '#F20404',
    },
    logger = console,
    actionFilter = () => true,
  } = options;

  if (!enabled) return config(set, get, api);

  return config(
    (...args) => {
      const prevState = get();

      // Determine action name
      const actionName =
        typeof args[1] === 'string'
          ? args[1]
          : typeof args[0] === 'function'
            ? args[0].name || 'anonymous'
            : 'set';

      if (!actionFilter(actionName)) {
        set(...args);
        return;
      }

      // Log start
      if (collapsed) {
        logger.groupCollapsed(
          `%caction %c${actionName} %c@ ${new Date().toLocaleTimeString()}`,
          `color: ${colors.title}; font-weight: bold;`,
          `color: ${colors.action}; font-weight: bold;`,
          `color: ${colors.title};`
        );
      } else {
        logger.group(
          `%caction %c${actionName}`,
          `color: ${colors.title};`,
          `color: ${colors.action}; font-weight: bold;`
        );
      }

      logger.log(
        '%cprev state',
        `color: ${colors.prev}; font-weight: bold;`,
        prevState
      );

      // Execute set
      set(...args);
      const nextState = get();

      // Log result
      logger.log(
        '%cnext state',
        `color: ${colors.next}; font-weight: bold;`,
        nextState
      );

      // Show diff
      const diff = {};
      const allKeys = new Set([
        ...Object.keys(prevState),
        ...Object.keys(nextState),
      ]);
      for (const key of allKeys) {
        if (prevState[key] !== nextState[key]) {
          diff[key] = {
            from: prevState[key],
            to: nextState[key],
          };
        }
      }
      if (Object.keys(diff).length > 0) {
        logger.log(
          '%cdiff',
          'color: #FF9800; font-weight: bold;',
          diff
        );
      }

      logger.groupEnd();
    },
    get,
    api
  );
};

// Usage
const useStore = create(
  comprehensiveLogger(
    (set) => ({
      count: 0,
      increment: () => set((s) => ({ count: s.count + 1 }), false, 'count/inc'),
    }),
    {
      enabled: true,
      collapsed: false,
      actionFilter: (name) => !name.includes('ui/'), // Exclude UI actions
    }
  )
);
```

### Undo/Redo Middleware

```jsx
// middleware/undoRedo.js
const undoRedo = (config, options = {}) => (set, get, api) => {
  const { limit = 50, include = [], exclude = [] } = options;

  const history = {
    past: [],
    future: [],
    isUndoing: false,
    isRedoing: false,
  };

  const shouldTrack = (actionName) => {
    if (exclude.length && exclude.some((p) => actionName.includes(p)))
      return false;
    if (include.length) return include.some((p) => actionName.includes(p));
    return true;
  };

  return config(
    (...args) => {
      const actionName = typeof args[1] === 'string' ? args[1] : '';

      // Don't track undo/redo operations
      if (history.isUndoing || history.isRedoing) {
        set(...args);
        return;
      }

      const currentState = get();
      set(...args);

      // Track in history
      if (shouldTrack(actionName)) {
        history.past.push(currentState);
        if (history.past.length > limit) history.past.shift();
        history.future = []; // Clear future on new action
      }
    },
    get,
    {
      ...api,
      undo: () => {
        if (history.past.length === 0) return;

        history.isUndoing = true;
        const previous = history.past.pop();
        const current = get();
        history.future.push(current);

        // Replace entire state
        set(previous, true);
        history.isUndoing = false;
      },
      redo: () => {
        if (history.future.length === 0) return;

        history.isRedoing = true;
        const next = history.future.pop();
        const current = get();
        history.past.push(current);

        set(next, true);
        history.isRedoing = false;
      },
      canUndo: () => history.past.length > 0,
      canRedo: () => history.future.length > 0,
    }
  );
};

// Usage
const useStore = create(
  undoRedo(
    (set) => ({
      text: '',
      setText: (text) => set({ text }, false, 'text/set'),
    }),
    { limit: 100, exclude: ['selection/'] }
  )
);

// Component
function TextEditor() {
  const text = useStore((s) => s.text);
  const setText = useStore((s) => s.setText);

  // Access undo/redo from the API (not state)
  const handleUndo = () => useStore.getState().undo?.();
  const handleRedo = () => useStore.getState().redo?.();

  return (
    <div>
      <textarea value={text} onChange={(e) => setText(e.target.value)} />
      <button onClick={handleUndo}>Undo</button>
      <button onClick={handleRedo}>Redo</button>
    </div>
  );
}
```

## 17. Zustand with React Compiler

### Understanding React Compiler + Zustand

React 19's Compiler automatically memoizes components. When combined with Zustand's efficient selectors, you get optimal performance with minimal boilerplate.

```
┌─────────────────────────────────────────────────────────────┐
│               React 19 Compiler                             │
│                                                             │
│  Component                                                  │
│     │                                                       │
│     │  Selector: (state) => state.count                    │
│     ▼                                                       │
│  ┌──────────────────────┐                                   │
│  │  Auto-memoization    │ ← Compiler adds memoization       │
│  │  (no manual useMemo) │   automatically                   │
│  └──────────────────────┘                                   │
│     │                                                       │
│     │  useStore(selector)                                    │
│     ▼                                                       │
│  Zustand (built-in selector comparison)                     │
│     │                                                       │
│     ▼                                                       │
│  Re-render only if selected value changes                   │
└─────────────────────────────────────────────────────────────┘
```

### SSR-Safe Store Creation

```jsx
// stores/createSSRStore.js
import { create } from 'zustand';

/**
 * Creates a Zustand store that works safely with SSR
 * Each request gets its own store instance
 */
export function createSSRStore(storeFactory) {
  // On server: create new store per request
  // On client: use global singleton
  const isServer = typeof window === 'undefined';

  if (isServer) {
    return () => storeFactory();
  }

  // Client: singleton
  let clientStore;
  return () => {
    if (!clientStore) {
      clientStore = storeFactory();
    }
    return clientStore;
  };
}

// Usage
const createProductStore = () =>
  create((set) => ({
    products: [],
    setProducts: (products) => set({ products }),
  }));

export const useProductStore = createSSRStore(createProductStore)();
```

### State Hydration Pattern

```jsx
// providers/StoreHydration.jsx
'use client';

import { useEffect, useState } from 'react';
import { useProductStore } from '../stores/useProductStore';
import { useCartStore } from '../stores/useCartStore';

/**
 * Hydrates Zustand stores from server data
 * Place this component high in your tree (e.g., in layout)
 */
export default function StoreHydration({ serverData }) {
  const [hydrated, setHydrated] = useState(false);

  useEffect(() => {
    // Hydrate stores with server data
    if (serverData.products) {
      useProductStore.setState({
        products: serverData.products,
        initialized: true,
      });
    }

    if (serverData.cart) {
      useCartStore.setState({ items: serverData.cart });
    }

    setHydrated(true);
  }, []);

  // Optional: Show nothing until hydrated, or a skeleton
  if (!hydrated) {
    return null; // Or <Skeleton />
  }

  return null; // This is a logic-only component
}

// Server layout provides data
// app/layout.jsx
export default async function RootLayout({ children }) {
  const [products, cart] = await Promise.all([
    fetchProducts(),
    fetchCart(), // If user is authenticated
  ]);

  return (
    <html>
      <body>
        <StoreHydration
          serverData={{ products, cart }}
        />
        {children}
      </body>
    </html>
  );
}
```

### Custom Storage Adapters

```jsx
// storage/customStorage.js

/**
 * localStorage adapter with encryption
 */
const encryptedStorage = {
  getItem: (name) => {
    const encrypted = localStorage.getItem(name);
    if (!encrypted) return null;

    try {
      // Decrypt (using a library like crypto-js)
      const decrypted = decrypt(encrypted, process.env.VITE_STORAGE_KEY);
      return decrypted;
    } catch {
      // If decryption fails, remove corrupted data
      localStorage.removeItem(name);
      return null;
    }
  },

  setItem: (name, value) => {
    const encrypted = encrypt(value, process.env.VITE_STORAGE_KEY);
    localStorage.setItem(name, encrypted);
  },

  removeItem: (name) => {
    localStorage.removeItem(name);
  },
};

/**
 * sessionStorage adapter
 */
const sessionStorage = {
  getItem: (name) => window.sessionStorage.getItem(name),
  setItem: (name, value) => window.sessionStorage.setItem(name, value),
  removeItem: (name) => window.sessionStorage.removeItem(name),
};

/**
 * Cookie storage adapter
 */
const cookieStorage = {
  getItem: (name) => {
    const match = document.cookie.match(new RegExp(`${name}=([^;]+)`));
    return match ? decodeURIComponent(match[1]) : null;
  },

  setItem: (name, value, days = 7) => {
    const expires = new Date(Date.now() + days * 864e5).toUTCString();
    document.cookie = `${name}=${encodeURIComponent(value)};expires=${expires};path=/;SameSite=Strict`;
  },

  removeItem: (name) => {
    document.cookie = `${name}=;expires=Thu, 01 Jan 1970 00:00:00 GMT;path=/`;
  },
};

/**
 * IndexedDB adapter for large data
 */
const indexedDBStorage = {
  db: null,

  async getDB() {
    if (this.db) return this.db;

    return new Promise((resolve, reject) => {
      const request = indexedDB.open('ZustandStore', 1);
      request.onerror = () => reject(request.error);
      request.onsuccess = () => {
        this.db = request.result;
        resolve(this.db);
      };
      request.onupgradeneeded = (event) => {
        event.target.result.createObjectStore('data');
      };
    });
  },

  async getItem(name) {
    const db = await this.getDB();
    return new Promise((resolve, reject) => {
      const tx = db.transaction('data', 'readonly');
      const store = tx.objectStore('data');
      const request = store.get(name);
      request.onsuccess = () => resolve(request.result || null);
      request.onerror = () => reject(request.error);
    });
  },

  async setItem(name, value) {
    const db = await this.getDB();
    return new Promise((resolve, reject) => {
      const tx = db.transaction('data', 'readwrite');
      const store = tx.objectStore('data');
      const request = store.put(value, name);
      request.onsuccess = () => resolve();
      request.onerror = () => reject(request.error);
    });
  },

  async removeItem(name) {
    const db = await this.getDB();
    return new Promise((resolve, reject) => {
      const tx = db.transaction('data', 'readwrite');
      const store = tx.objectStore('data');
      const request = store.delete(name);
      request.onsuccess = () => resolve();
      request.onerror = () => reject(request.error);
    });
  },
};
```

### Advanced Persist Configuration

```jsx
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

const useAppStore = create(
  persist(
    (set, get) => ({
      // State
      user: null,
      preferences: { theme: 'light', lang: 'en' },
      cart: { items: [], lastUpdated: null },
      drafts: {},
      cache: {},

      // Actions
      setUser: (user) => set({ user }),
      updatePreferences: (prefs) =>
        set((state) => ({
          preferences: { ...state.preferences, ...prefs },
        })),
    }),
    {
      name: 'app-v2-storage',

      // ─── Custom Storage ───
      // storage: encryptedStorage,  // Encrypt sensitive data
      // storage: sessionStorage,    // Only for session
      // storage: indexedDBStorage,  // Large data

      // ─── Partial Persistence ───
      // Only persist specific fields
      partialize: (state) => ({
        user: state.user,
        preferences: state.preferences,
        // Don't persist: cart, drafts, cache
      }),

      // ─── Transform (serialize/deserialize) ───
      serialize: (state) => {
        return JSON.stringify({
          ...state,
          timestamp: Date.now(),
        });
      },

      deserialize: (str) => {
        const parsed = JSON.parse(str);
        // Check if data is stale (> 7 days)
        const age = Date.now() - (parsed.timestamp || 0);
        if (age > 7 * 24 * 60 * 60 * 1000) {
          return { state: null, version: parsed.version };
        }
        return parsed;
      },

      // ─── Version Migration ───
      version: 2,

      migrate: (persistedState, version) => {
        // v0 → v1: Added preferences object
        if (version === 0) {
          return {
            ...persistedState,
            preferences: { theme: 'light', lang: 'en' },
          };
        }

        // v1 → v2: Renamed 'settings' to 'preferences'
        if (version === 1) {
          const { settings, ...rest } = persistedState;
          return {
            ...rest,
            preferences: settings || { theme: 'light', lang: 'en' },
          };
        }

        return persistedState;
      },

      // ─── Merge Strategy ───
      // How to merge persisted state with initial state
      merge: (persistedState, currentState) => {
        return {
          ...currentState,
          ...persistedState.state,
          // Always use current state for these
          cache: currentState.cache,
          drafts: currentState.drafts,
        };
      },

      // ─── On Rehydration ───
      onRehydrateStorage: () => (state, error) => {
        if (error) {
          console.error('Failed to rehydrate store:', error);
        } else {
          console.log('Store rehydrated successfully');
          state?.onHydrated?.();
        }
      },
    }
  )
);
```

```jsx
const OrderStates = {
  IDLE: 'idle',
  SELECTING: 'selecting',
  CHECKOUT: 'checkout',
  PROCESSING: 'processing',
  CONFIRMED: 'confirmed',
  FAILED: 'failed',
};

// Define valid transitions
const transitions = {
  [OrderStates.IDLE]: ['SELECTING'],
  [OrderStates.SELECTING]: ['CHECKOUT', 'IDLE'],
  [OrderStates.CHECKOUT]: ['PROCESSING', 'SELECTING'],
  [OrderStates.PROCESSING]: ['CONFIRMED', 'FAILED'],
  [OrderStates.CONFIRMED]: ['IDLE'],
  [OrderStates.FAILED]: ['CHECKOUT', 'IDLE'],
};

export const useOrderMachine = create((set, get) => ({
  // Current state
  state: OrderStates.IDLE,
  previousState: null,
  context: {
    items: [],
    shippingAddress: null,
    paymentMethod: null,
    error: null,
    orderId: null,
  },

  // Check if transition is valid
  canTransition: (event) => {
    const { state: currentState } = get();
    return transitions[currentState]?.includes(event) || false;
  },

  // Transition to new state
  transition: (event, contextUpdates = {}) => {

    if (!transitions[currentState]?.includes(event)) {
      console.warn(
        `Invalid transition: ${currentState} → ${event}`
      );
      return false;
    }

    set({
      previousState: currentState,
      state: OrderStates[event] || event,
      context: {
        ...get().context,
        ...contextUpdates,
      },
    });

    return true;
  },

    set((s) => ({ context: { ...s.context, items } })),
  setShippingAddress: (address) =>
    set((s) => ({ context: { ...s.context, shippingAddress: address } })),

    if (context.items.length === 0) return false;
    return get().transition('CHECKOUT');

        shippingAddress: context.shippingAddress,
        paymentMethod: context.paymentMethod,

      return order;
    } catch (error) {
      transition('FAILED', { error: error.message });
      throw error;
    }
  },

    set({
      state: OrderStates.IDLE,
      previousState: null,
      context: {
        items: [],
        shippingAddress: null,
        paymentMethod: null,
        error: null,
        orderId: null,

    },
```

```jsx
        guard: (ctx) => ctx.retryCount < 3,
      },
      GIVE_UP: { target: 'unauthenticated' },
    },
  },

  dispatch: (actionType, payload) => {
    const { state, context, transitions } = get();
    const transition = transitions[state]?.[actionType];

    if (!transition) {
      console.warn(`No transition for ${state}.${actionType}`);
      return false;
    }

    // Check guard
    if (transition.guard && !transition.guard(context, payload)) {
      console.warn(`Guard failed for ${state}.${actionType}`);
      return false;
    }

    // Execute action if present
    const contextUpdates = transition.action
      ? transition.action(context, payload)
      : payload;

    set({
      state: transition.target,
      context: { ...context, ...contextUpdates },
    });

    return true;
  },

  // Convenience actions
  login: (credentials) => get().dispatch('LOGIN', credentials),
  logout: () => get().dispatch('LOGOUT'),
```

```jsx
// utils/storeDebugger.js
export function createStoreDebugger(storeName, store) {
  if (process.env.NODE_ENV === 'production') return {};

  const history = [];
  const maxHistory = 100;

  // Subscribe to all changes
  const unsubscribe = store.subscribe(
    (state) => state, // Select entire state
    (current, previous) => {
      const entry = {
        timestamp: Date.now(),
        previous,
        current,
        diff: computeDiff(previous, current),
      };

      history.push(entry);
      if (history.length > maxHistory) history.shift();
    }
  );

  return {
    getHistory: () => [...history],

    getHistoryForKey: (key) =>
      history
        .filter((h) => h.diff[key])
        .map((h) => ({
          timestamp: h.timestamp,
          from: h.diff[key].from,
          to: h.diff[key].to,
        })),

    getStateAt: (timestamp) => {
      const entry = history.find((h) => h.timestamp >= timestamp);
      return entry?.current;
    },

    replay: (index) => {
      if (history[index]) {
        store.setState(history[index].current, true);
      }
    },

    export: () => JSON.stringify(history, null, 2),

    import: (data) => {
      const parsed = JSON.parse(data);
      history.push(...parsed);
    },

    destroy: () => unsubscribe(),
  };
}

function computeDiff(prev, next) {
  const diff = {};
  const allKeys = new Set([...Object.keys(prev), ...Object.keys(next)]);
  for (const key of allKeys) {
    if (prev[key] !== next[key]) {
      diff[key] = { from: prev[key], to: next[key] };
    }
  }
  return diff;
}
```

## 23. Optimistic Updates Pattern

### Zustand + React 19 useOptimistic Bridge

```jsx
// React 19 provides useOptimistic, but Zustand can manage
// the full optimistic update lifecycle

// stores/useOptimisticStore.js
import { create } from 'zustand';

export const useOptimisticStore = create((set, get) => ({
  // ─── State ───
  items: [],
  pendingOperations: new Map(), // Track in-flight operations

  // ─── Optimistic Add ───
  optimisticAdd: (item, operationId) => {
    const tempId = `temp-${operationId}`;

    // 1. Immediately add to UI
    set((state) => ({
      items: [...state.items, { ...item, id: tempId, optimistic: true }],
      pendingOperations: new Map(state.pendingOperations).set(operationId, {
        type: 'add',
        tempId,
        timestamp: Date.now(),
      }),
    }));

    return tempId;
  },

  // ─── Confirm Operation ───
  confirmOperation: (operationId, serverItem) => {
    set((state) => {
      const pending = state.pendingOperations.get(operationId);
      if (!pending) return state;

      const newPending = new Map(state.pendingOperations);
      newPending.delete(operationId);

      // Replace temp item with server-confirmed item
      return {
        items: state.items.map((item) =>
          item.id === pending.tempId
            ? { ...serverItem, optimistic: false }
            : item
        ),
        pendingOperations: newPending,
      };
    });
  },

  // ─── Rollback Operation ───
  rollbackOperation: (operationId) => {
    set((state) => {
      const pending = state.pendingOperations.get(operationId);
      if (!pending) return state;

      const newPending = new Map(state.pendingOperations);
      newPending.delete(operationId);

      return {
        // Remove the optimistic item
        items: state.items.filter((item) => item.id !== pending.tempId),
        pendingOperations: newPending,
      };
    });
  },

  // ─── Optimistic Update ───
  optimisticUpdate: (id, updates, operationId) => {
    // Save original for rollback
    const originalItem = get().items.find((i) => i.id === id);

    set((state) => ({
      items: state.items.map((item) =>
        item.id === id ? { ...item, ...updates, optimistic: true } : item
      ),
      pendingOperations: new Map(state.pendingOperations).set(operationId, {
        type: 'update',
        id,
        original: originalItem,
        timestamp: Date.now(),
      }),
    }));
  },

  confirmUpdate: (operationId) => {
    set((state) => {
      const newPending = new Map(state.pendingOperations);
      newPending.delete(operationId);

      return {
        items: state.items.map((item) =>
          item.optimistic ? { ...item, optimistic: false } : item
        ),
        pendingOperations: newPending,
      };
    });
  },

  rollbackUpdate: (operationId) => {
    set((state) => {
      const pending = state.pendingOperations.get(operationId);
      if (!pending) return state;

      const newPending = new Map(state.pendingOperations);
      newPending.delete(operationId);

      return {
        items: state.items.map((item) =>
          item.id === pending.id ? pending.original : item
        ),
        pendingOperations: newPending,
      };
    });
  },

  // ─── Optimistic Delete ───
  optimisticDelete: (id, operationId) => {
    const originalItem = get().items.find((i) => i.id === id);

    set((state) => ({
      items: state.items.filter((item) => item.id !== id),
      pendingOperations: new Map(state.pendingOperations).set(operationId, {
        type: 'delete',
        id,
        original: originalItem,
        timestamp: Date.now(),
      }),
    }));
  },

  rollbackDelete: (operationId) => {
    set((state) => {
      const pending = state.pendingOperations.get(operationId);
      if (!pending) return state;

      const newPending = new Map(state.pendingOperations);
      newPending.delete(operationId);

      // Restore the deleted item
      return {
        items: [...state.items, pending.original],
        pendingOperations: newPending,
      };
    });
  },
}));
```

```tsx
    addItem: (item) =>

    getSelected: () =>

  }));
}
```

```tsx
        options.persist
          ? persist(initializer as any, options.persist) as any
          : initializer as any,
        options.devtools
      ) as any
    );
  }

  return store;
}

// Usage
const useMyStore = createAdvancedStore(
  (set) => ({
    value: 0,
    setValue: (v: number) => set({ value: v }),
  }),
  {
    devtools: { name: 'MyStore' },
    persist: { name: 'my-store', version: 1 },
  }
);
```

### State Hydration Pattern

```jsx
// stores/useHydration.js
import { useState, useEffect } from 'react';

export function useHydration(store) {
  const [hydrated, setHydrated] = useState(false);

  useEffect(() => {
    // Check if store has been rehydrated from storage
    const unsub = store.persist?.onFinishHydration?.(() => {
      setHydrated(true);
    });

    // If no persist or already hydrated
    if (!store.persist) {
      setHydrated(true);
    }

    return () => unsub?.();
  }, [store]);

  return hydrated;
}

// Usage in component
function CartDrawer() {
  const hydrated = useHydration(useCartStore);
  const items = useCartStore((state) => state.items);
  const isOpen = useCartStore((state) => state.isOpen);

  // Show skeleton until hydrated (prevents hydration mismatch)
  if (!hydrated) {
    return <CartSkeleton />;
  }

  return (
    <aside className={isOpen ? 'open' : ''}>
      <h2>Cart ({items.length})</h2>
      {items.map((item) => (
        <CartItem key={item.id} item={item} />
      ))}
    </aside>
  );
}
```

### Store Monitoring and Health Checks

```jsx
// utils/storeMonitor.js
class StoreMonitor {
  metrics = {
    setCalls: 0,
    subscriptions: 0,
    errors: [],
    performance: [],
  };

  attach(store, name) {
    const originalSet = store.setState;

    store.setState = (...args) => {
      const start = performance.now();
      this.metrics.setCalls++;

      try {
        originalSet.apply(store, args);
      } catch (error) {
        this.metrics.errors.push({
          timestamp: Date.now(),
          error: error.message,
          store: name,
        });
        throw error;
      }

      const duration = performance.now() - start;
      this.metrics.performance.push(duration);

      // Keep last 100 measurements
      if (this.metrics.performance.length > 100) {
        this.metrics.performance.shift();
      }
    };

    return this;
  }

  getReport() {
    const perf = this.metrics.performance;
    return {
      totalSetCalls: this.metrics.setCalls,
      averageSetTime: perf.length > 0
        ? perf.reduce((a, b) => a + b, 0) / perf.length
        : 0,
      maxSetTime: perf.length > 0 ? Math.max(...perf) : 0,
      recentErrors: this.metrics.errors.slice(-10),
    };
  }

  reset() {
    this.metrics = {
      setCalls: 0,
      subscriptions: 0,
      errors: [],
      performance: [],
    };
  }
}

export const storeMonitor = new StoreMonitor();

// Usage
if (process.env.NODE_ENV !== 'production') {
  storeMonitor.attach(useAuthStore, 'auth');
  storeMonitor.attach(useCartStore, 'cart');

  // Expose to window for debugging
  window.storeMonitor = storeMonitor;
  // In console: storeMonitor.getReport()
}
```

### Production Store Setup Template

```jsx
// stores/createProductionStore.js
import { create } from 'zustand';
import { devtools, persist, subscribeWithSelector } from 'zustand/middleware';
import { immer } from 'zustand/middleware/immer';

export function createProductionStore(name, config, options = {}) {
  const {
    persist: persistOptions,
    devtools: devtoolsOptions = { name },
    enableDevTools = process.env.NODE_ENV === 'development',
  } = options;

  let store = subscribeWithSelector(immer(config));

  if (persistOptions) {
    store = persist(store, {
      name: `${name}-storage`,
      ...persistOptions,
    });
  }

  if (enableDevTools) {
    store = devtools(store, devtoolsOptions);
  }

  return create(store, options);
}

// Usage

  'user',

    profile: null,
    preferences: {},

    setProfile: (profile) => set({ profile }),
    updatePreferences: (prefs) =>
      set((draft) => {
        draft.preferences = { ...draft.preferences, ...prefs };
      }),
  }),
  {
    persist: {
      partialize: (state) => ({
        preferences: state.preferences,
      }),
    },
  }
);
```
