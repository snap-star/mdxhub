## 6. The Slices Pattern

### What are Slices?

Slices are modular store definitions that compose into a single store. Each slice manages a domain of your application state.

### Basic Slice Pattern

```jsx
// stores/slices/counterSlice.js
export const createCounterSlice = (set, get) => ({
  count: 0,
  increment: () => set((state) => ({ count: state.count + 1 })),
  decrement: () => set((state) => ({ count: state.count - 1 })),
  reset: () => set({ count: 0 }),
  setCount: (count) => set({ count }),
});

// stores/slices/userSlice.js
export const createUserSlice = (set, get) => ({
  user: null,
  isAuthenticated: false,
  isLoading: false,
  
  login: async (credentials) => {
    set({ isLoading: true });
    try {
      const user = await api.login(credentials);
      set({ user, isAuthenticated: true, isLoading: false });
    } catch (error) {
      set({ isLoading: false, error: error.message });
    }
  },
  
  logout: () => {
    set({ user: null, isAuthenticated: false });
    api.logout();
  },
  
  updateProfile: (updates) =>
    set((state) => ({
      user: state.user ? { ...state.user, ...updates } : null,
    })),
});

// stores/slices/cartSlice.js
export const createCartSlice = (set, get) => ({
  items: [],
  isOpen: false,
  
  addItem: (product) =>
    set((state) => {
      const existing = state.items.find((i) => i.id === product.id);
      if (existing) {
        return {
          items: state.items.map((i) =>
            i.id === product.id
              ? { ...i, quantity: i.quantity + 1 }
              : i
          ),
        };
      }
      return { items: [...state.items, { ...product, quantity: 1 }] };
    }),
  
  removeItem: (productId) =>
    set((state) => ({
      items: state.items.filter((i) => i.id !== productId),
    })),
  
  updateQuantity: (productId, quantity) =>
    set((state) => ({
      items: quantity <= 0
        ? state.items.filter((i) => i.id !== productId)
        : state.items.map((i) =>
            i.id === productId ? { ...i, quantity } : i
          ),
    })),
  
  clearCart: () => set({ items: [] }),
  toggleCart: () => set((state) => ({ isOpen: !state.isOpen })),
  
  // Computed value as a function (not reactive)
  getTotalItems: () =>
    get().items.reduce((sum, i) => sum + i.quantity, 0),
  
  getTotalPrice: () =>
    get().items.reduce((sum, i) => sum + i.price * i.quantity, 0),
});
```

```jsx
// stores/useAppStore.js
```

### Slices with Dependencies

```jsx
// stores/slices/orderSlice.js
// This slice depends on cart state
export const createOrderSlice = (set, get) => ({
  orderStatus: 'idle', // idle | placing | success | error
  currentOrder: null,
  
  placeOrder: async (shippingAddress) => {
    const { items, getTotalPrice, clearCart } = get();
    
    if (items.length === 0) {
      set({ orderStatus: 'error', error: 'Cart is empty' });
      return;
    }
    
    set({ orderStatus: 'placing' });
    
    try {
      const order = await api.createOrder({
        items,
        total: getTotalPrice(),
        shippingAddress,
      });
      
      set({ orderStatus: 'success', currentOrder: order });
      clearCart(); // Cross-slice action call
    } catch (error) {
      set({ orderStatus: 'error', error: error.message });
    }
  },
  
  resetOrder: () =>
    set({ orderStatus: 'idle', currentOrder: null, error: null }),
});
```

## 7. Middleware Deep Dive

### What is Middleware?

Middleware wraps Zustand's store to add behavior — logging, persistence, devtools, etc.

```
┌─────────────┐    ┌──────────────┐    ┌──────────────┐    ┌──────────┐
│  Component  │───▶│  Middleware  │───▶│  Middleware  │───▶│   Core   │
│   calls     │    │  (devtools)  │    │  (persist)   │    │  Store   │
│   action    │    │              │    │              │    │          │
└─────────────┘    └──────────────┘    └──────────────┘    └──────────┘
                                                         ▲
                                              set() / get()
```

### devtools Middleware

```jsx
import { create } from 'zustand';
import { devtools } from 'zustand/middleware';

const useStore = create(
  devtools(
    (set, get) => ({
      count: 0,
      increment: () =>
        set(
          (state) => ({ count: state.count + 1 }),
          false, // Not replacing all state
          'count/increment' // Action name in Redux DevTools
        ),
      decrement: () =>
        set(
          (state) => ({ count: state.count - 1 }),
          false,
          'count/decrement'
        ),
    }),
    {
      name: 'MyAppStore', // Store name in DevTools
      enabled: process.env.NODE_ENV === 'development',
    }
  )
);
```

### persist Middleware

```jsx
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

const useStore = create(
  persist(
    (set, get) => ({
      theme: 'light',
      sidebarOpen: true,
      userPreferences: {
        fontSize: 14,
        reducedMotion: false,
      },
      
      setTheme: (theme) => set({ theme }),
      toggleSidebar: () =>
        set((state) => ({ sidebarOpen: !state.sidebarOpen })),
    }),
    {
      name: 'my-app-storage', // localStorage key
      
      // Only persist specific fields
      partialize: (state) => ({
        theme: state.theme,
        userPreferences: state.userPreferences,
        // Don't persist sidebarOpen — reset on refresh
      }),
      
      // Custom version for migrations
      version: 1,
      
      // Migrate old stored data
      migrate: (persistedState, version) => {
        if (version === 0) {
          // Migrate from v0 to v1
          return {
            ...persistedState,
            userPreferences: {
              fontSize: 14,
              reducedMotion: false,
              ...persistedState.userPreferences,
            },
          };
        }
        return persistedState;
      },
    }
  )
);
```

### immer Middleware

```jsx
import { create } from 'zustand';
import { immer } from 'zustand/middleware/immer';

// Without immer — verbose nested updates
const useStoreVerbose = create((set) => ({
  user: { profile: { settings: { theme: 'dark' } } },
  setTheme: (theme) =>
    set((state) => ({
      user: {
        ...state.user,
        profile: {
          ...state.user.profile,
          settings: {
            ...state.user.profile.settings,
            theme,
          },
        },
      },
    })),
}));

// With immer — mutate directly!
const useStore = create(
  immer((set) => ({
    user: { profile: { settings: { theme: 'dark' } } },
    notifications: [],
    
    setTheme: (theme) =>
      set((draft) => {
        draft.user.profile.settings.theme = theme; // Direct mutation!
      }),
    
    addNotification: (notification) =>
      set((draft) => {
        draft.notifications.push(notification); // Array push works!
      }),
    
    markNotificationRead: (id) =>
      set((draft) => {
        const notif = draft.notifications.find((n) => n.id === id);
        if (notif) notif.read = true; // Direct property assignment!
      }),
    
    removeNotification: (id) =>
      set((draft) => {
        const index = draft.notifications.findIndex((n) => n.id === id);
        if (index !== -1) draft.notifications.splice(index, 1);
      }),
  }))
);
```

### subscribeWithSelector Middleware

```jsx
import { create } from 'zustand';
import { subscribeWithSelector } from 'zustand/middleware';

const useStore = create(
  subscribeWithSelector((set) => ({
    count: 0,
    name: 'Zustand',
    theme: 'dark',
    
    setCount: (count) => set({ count }),
    setName: (name) => set({ name }),
    setTheme: (theme) => set({ theme }),
  }))
);

// Subscribe to specific selector (efficient!)
const unsub = useStore.subscribe(
  (state) => state.count,
  (count, prevCount) => {
    console.log(`Count: ${prevCount} → ${count}`);
  },
  { equalityFn: (a, b) => a === b } // Optional custom equality
);

// Subscribe to equality change (fires only when result changes)
useStore.subscribe(
  (state) => ({ count: state.count, name: state.name }),
  (result) => {
    console.log('Count or name changed:', result);
  },
  { equalityFn: shallow }
);
```

### Combining Multiple Middleware

```jsx
import { create } from 'zustand';
import { devtools, persist, immer, subscribeWithSelector } from 'zustand/middleware';

const useStore = create(
  devtools(
    persist(
      subscribeWithSelector(
        immer((set, get) => ({
          // Your store definition
          count: 0,
          user: null,
          
          increment: () =>
            set((draft) => {
              draft.count += 1;
            }),
          
          login: async (credentials) => {
            const user = await api.login(credentials);
            set((draft) => {
              draft.user = user;
            });
          },
        }))
      ),
      {
        name: 'app-storage',
        partialize: (state) => ({ user: state.user }),
      }
    ),
    { name: 'MyStore' }
  )
);
```

**Middleware Order Matters:**

```
// Outside → Inside
// devtools sees everything (outermost)
// persist intercepts set for saving
// subscribeWithSelector adds subscription features
// immer processes mutations
// Your store (innermost)

devtools(
  persist(
    subscribeWithSelector(
      immer(store)
    )
  )
)
```

## 8. Async Operations in Zustand

### Thunk Pattern

```jsx
import { create } from 'zustand';

const useProductStore = create((set, get) => ({
  // ─── State ───
  products: [],
  selectedProduct: null,
  isLoading: false,
  error: null,
  
  // ─── Simple sync actions ───
  setError: (error) => set({ error }),
  clearError: () => set({ error: null }),
  
  // ─── Async thunks ───
  fetchProducts: async (category) => {
    set({ isLoading: true, error: null });
    
    try {
      const products = await api.getProducts(category);
      set({ products, isLoading: false });
    } catch (error) {
      set({ error: error.message, isLoading: false });
    }
  },
  
  fetchProductById: async (id) => {
    set({ isLoading: true, error: null, selectedProduct: null });
    
    try {
      const product = await api.getProduct(id);
      set({ selectedProduct: product, isLoading: false });
    } catch (error) {
      set({ error: error.message, isLoading: false });
    }
  },
  
  createProduct: async (productData) => {
    set({ isLoading: true, error: null });
    
    try {
      const newProduct = await api.createProduct(productData);
      set((state) => ({
        products: [...state.products, newProduct],
        isLoading: false,
      }));
      return newProduct; // Return for caller to use
    } catch (error) {
      set({ error: error.message, isLoading: false });
      throw error; // Re-throw for error handling
    }
  },
  
  updateProduct: async (id, updates) => {
    set({ isLoading: true, error: null });
    
    try {
      const updated = await api.updateProduct(id, updates);
      set((state) => ({
        products: state.products.map((p) =>
          p.id === id ? updated : p
        ),
        selectedProduct:
          state.selectedProduct?.id === id
            ? updated
            : state.selectedProduct,
        isLoading: false,
      }));
    } catch (error) {
      set({ error: error.message, isLoading: false });
    }
  },
  
  deleteProduct: async (id) => {
    try {
      await api.deleteProduct(id);
      set((state) => ({
        products: state.products.filter((p) => p.id !== id),
        selectedProduct:
          state.selectedProduct?.id === id
            ? null
            : state.selectedProduct,
      }));
    } catch (error) {
      set({ error: error.message });
    }
  },
}));
```

### Request Status Pattern

```jsx
// More granular async state tracking
const useAsyncStore = create((set, get) => ({
  // Map of request statuses by key
  requests: {},
  
  // Get status of a specific request
  getRequestStatus: (key) =>
    get().requests[key] || { status: 'idle', data: null, error: null },
  
  // Execute an async operation with status tracking
  execute: async (key, asyncFn) => {
    set((state) => ({
      requests: {
        ...state.requests,
        [key]: { status: 'loading', data: null, error: null },
      },
    }));
    
    try {
      const data = await asyncFn();
      set((state) => ({
        requests: {
          ...state.requests,
          [key]: { status: 'success', data, error: null },
        },
      }));
      return data;
    } catch (error) {
      set((state) => ({
        requests: {
          ...state.requests,
          [key]: { status: 'error', data: null, error: error.message },
        },
      }));
      throw error;
    }
  },
  
  // Clear a specific request
  clearRequest: (key) =>
    set((state) => {
      const { [key]: _, ...rest } = state.requests;
      return { requests: rest };
    }),
}));

// Usage in component
function ProductList({ category }) {
  const { execute, getRequestStatus, clearRequest } = useAsyncStore();
  const requestKey = `products-${category}`;
  const { status, data: products, error } = getRequestStatus(requestKey);
  
  useEffect(() => {
    execute(requestKey, () => api.getProducts(category));
    return () => clearRequest(requestKey);
  }, [category]);
  
  if (status === 'loading') return <Skeleton />;
  if (status === 'error') return <ErrorMessage error={error} />;
  
  return (
    <ul>
      {products?.map((p) => <ProductCard key={p.id} product={p} />)}
    </ul>
  );
}
```

### Cancellation Pattern

```jsx
import { create } from 'zustand';

const useSearchStore = create((set, get) => ({
  results: [],
  isSearching: false,
  abortController: null,
  
  search: async (query) => {
    // Cancel previous search
    const prevController = get().abortController;
    if (prevController) {
      prevController.abort();
    }
    
    if (!query.trim()) {
      set({ results: [], isSearching: false, abortController: null });
      return;
    }
    
    const controller = new AbortController();
    set({ isSearching: true, abortController: controller });
    
    try {
      const results = await api.search(query, {
        signal: controller.signal,
      });
      
      // Only update if this request wasn't aborted
      if (!controller.signal.aborted) {
        set({ results, isSearching: false, abortController: null });
      }
    } catch (error) {
      if (error.name !== 'AbortError') {
        set({ isSearching: false, abortController: null, error });
      }
    }
  },
}));
```

## 10. Testing Zustand Stores

### Testing Store Logic

```jsx
// stores/useCounterStore.test.js
import { describe, it, expect, beforeEach } from 'vitest';
import { useCounterStore } from './useCounterStore';

describe('useCounterStore', () => {
  // Reset store before each test
  beforeEach(() => {
    useCounterStore.setState({
      count: 0,
      step: 1,
    });
  });

  it('should initialize with default values', () => {
    const state = useCounterStore.getState();
    expect(state.count).toBe(0);
    expect(state.step).toBe(1);
  });

    useCounterStore.getState().increment();
    expect(useCounterStore.getState().count).toBe(1);
  });

  it('should decrement count', () => {
    useCounterStore.setState({ count: 5 });
    useCounterStore.getState().decrement();
    expect(useCounterStore.getState().count).toBe(4);
  });

  it('should respect step value', () => {
    useCounterStore.setState({ step: 5 });
    useCounterStore.getState().increment();
    expect(useCounterStore.getState().count).toBe(5);
  });

  it('should not exceed max', () => {
    useCounterStore.setState({ count: 99, step: 5, max: 100 });
    useCounterStore.getState().increment();
    expect(useCounterStore.getState().count).toBe(99); // Blocked by max
  });

  it('should reset to zero', () => {
    useCounterStore.setState({ count: 42 });
    useCounterStore.getState().reset();
    expect(useCounterStore.getState().count).toBe(0);
  });
});
```

```jsx
// utils/eventBus.js
const listeners = new Map();

export const eventBus = {
  on: (event, callback) => {
    if (!listeners.has(event)) listeners.set(event, new Set());
    listeners.get(event).add(callback);
    return () => listeners.get(event)?.delete(callback);
  },
  emit: (event, data) => {
    listeners.get(event)?.forEach((cb) => cb(data));
  },
```

```jsx
    });
```
```jsx
  },
```

```jsx
// middleware/analytics.js
const analytics = (config, options = {}) => (set, get, api) => {
  const { trackEvent = console.log, exclude = [] } = options;
  
  return config(
    (...args) => {
      // Get action name if provided (devtools format)
      const actionName = typeof args[1] === 'string' ? args[1] : 'unknown';
      
      if (!exclude.includes(actionName)) {
        const prevState = get();
        set(...args);
        const nextState = get();
        
        // Track state change
        trackEvent('state_change', {
          action: actionName,
          diff: getDiff(prevState, nextState),
          timestamp: Date.now(),
        });
      } else {
        set(...args);
      }
    },
    get,
    api
  );
};

// Helper to compute state diff
function getDiff(prev, next) {
  const diff = {};
  for (const key of new Set([...Object.keys(prev), ...Object.keys(next)])) {
    if (prev[key] !== next[key]) {
      diff[key] = { from: prev[key], to: next[key] };
    }
  }
  return diff;
}

      exclude: ['ui/toggle'], // Don't track UI toggles
```

```jsx
  return config(
    (...args) => {
      // Get the state update

        ? args[0](get())
```

```jsx
import { create } from 'zustand';
import { devtools, persist, immer } from 'zustand/middleware';

// Correct order: devtools (outermost) → persist → immer → store (innermost)
const useStore = create(
  devtools(
    persist(
      immer((set, get) => ({
        // Store definition
        items: [],
        addItem: (item) =>
          set((draft) => {
            draft.items.push(item);
          }),
      })),
      {
        name: 'my-app-storage',
        partialize: (state) => ({ items: state.items }),
      }
    ),
    { name: 'MyStore' }
  )
);

// Execution flow for set():
// 1. devtools: Log action name, track timing
// 2. persist: Intercept set, save to storage
// 3. immer: Convert mutations to immutable updates
// 4. store: Apply the actual state change
```

### Redux → Zustand Migration

```jsx
// ─── BEFORE: Redux ───

// store.js
import { configureStore, createSlice } from '@reduxjs/toolkit';

const counterSlice = createSlice({
  name: 'counter',
  initialState: { value: 0 },
  reducers: {
    increment: (state) => { state.value += 1; },
    decrement: (state) => { state.value -= 1; },
    incrementByAmount: (state, action) => {
      state.value += action.payload;
    },
  },
});

export const { increment, decrement, incrementByAmount } = counterSlice.actions;

export const store = configureStore({
  reducer: {
    counter: counterSlice.reducer,
  },
});

// Component
import { useSelector, useDispatch } from 'react-redux';

function Counter() {
  const count = useSelector((state) => state.counter.value);
  const dispatch = useDispatch();
  
  return (
    <div>
      <button onClick={() => dispatch(increment())}>+</button>
      <span>{count}</span>
      <button onClick={() => dispatch(decrement())}>-</button>
    </div>
  );
}

// ─── AFTER: Zustand ───

// stores/useCounterStore.js
import { create } from 'zustand';

export const useCounterStore = create((set) => ({
  value: 0,
  increment: () => set((state) => ({ value: state.value + 1 })),
  decrement: () => set((state) => ({ value: state.value - 1 })),
  incrementByAmount: (amount) =>
    set((state) => ({ value: state.value + amount })),
}));

// Component
function Counter() {
  const count = useCounterStore((state) => state.value);
  const increment = useCounterStore((state) => state.increment);
  const decrement = useCounterStore((state) => state.decrement);

    <div>
      <button onClick={increment}>+</button>

}
```

### Context API → Zustand Migration

```jsx
// ─── BEFORE: Context API ───

// contexts/ThemeContext.js
import { createContext, useContext, useState } from 'react';

const ThemeContext = createContext(null);

export function ThemeProvider({ children }) {
  const [theme, setTheme] = useState('light');
  const toggleTheme = () =>
    setTheme((t) => (t === 'light' ? 'dark' : 'light'));
  
  return (
    <ThemeContext.Provider value={{ theme, setTheme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

export const useTheme = () => {
  const ctx = useContext(ThemeContext);
  if (!ctx) throw new Error('Missing ThemeProvider');
  return ctx;
};

// App.js — wrap with providers
<ThemeProvider>
  <AuthProvider>
    <CartProvider>
      <App />
    </CartProvider>
  </AuthProvider>
</ThemeProvider>

// ─── AFTER: Zustand ───

// stores/useThemeStore.js
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

export const useThemeStore = create(
  persist(
    (set) => ({
      theme: 'light',
      toggleTheme: () =>
        set((state) => ({
          theme: state.theme === 'light' ? 'dark' : 'light',
        })),
      setTheme: (theme) => set({ theme }),
    }),
    { name: 'theme-storage' }
  )
);

// App.js — no providers needed!
<App /> // Zustand hooks work anywhere

// Component — simpler
function ThemeToggle() {
  const theme = useThemeStore((state) => state.theme);
  const toggleTheme = useThemeStore((state) => state.toggleTheme);
  
  return (
    <button onClick={toggleTheme}>
      Switch to {theme === 'light' ? 'dark' : 'light'}
    </button>
  );
}
```

### Gradual Migration Strategy

```jsx
// Step 1: Create Zustand store alongside Redux
// Step 2: Move one feature at a time
// Step 3: Use adapter pattern during transition

// adapters/zustandToRedux.js
import { useCounterStore } from '../stores/useCounterStore';

// Adapter: Make Zustand look like Redux for gradual migration
export function useCounterAdapter() {
  const count = useCounterStore((state) => state.count);
  const increment = useCounterStore((state) => state.increment);
  const decrement = useCounterStore((state) => state.decrement);
  
  return {
    // Redux-style API
    state: { counter: { value: count } },
    dispatch: (action) => {
      switch (action.type) {
        case 'counter/increment':
          increment();
          break;
        case 'counter/decrement':
          decrement();
          break;
      }
    },
  };
}

// Old component still works during migration
function LegacyCounter() {
  const { state, dispatch } = useCounterAdapter();
  
  return (
    <div>
      <button onClick={() => dispatch({ type: 'counter/decrement' })}>-</button>
      <span>{state.counter.value}</span>
      <button onClick={() => dispatch({ type: 'counter/increment' })}>+</button>
    </div>
  );
}
```
