## Section 4: State Management with useState

### Understanding useState

The `useState` hook lets you add state to function components. State causes re-renders when updated.

```jsx
import { useState } from 'react';

function Counter() {
  // useState returns: [currentValue, setterFunction]
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>Increment</button>
      <button onClick={() => setCount(count - 1)}>Decrement</button>
      <button onClick={() => setCount(0)}>Reset</button>
    </div>
  );
}
```

### State with Different Data Types

```jsx
function UserProfile() {
  // String state
  const [name, setName] = useState('Alice');

  // Number state
  const [age, setAge] = useState(25);

  // Boolean state
  const [isEditing, setIsEditing] = useState(false);

  // Array state
  const [hobbies, setHobbies] = useState(['Reading', 'Coding']);

  // Object state
  const [user, setUser] = useState({
    firstName: 'Alice',
    lastName: 'Johnson',
    email: 'alice@example.com',
  });

  // Updating object state (must spread existing properties)
  const updateEmail = (newEmail) => {
    setUser({ ...user, email: newEmail });
  };

  // Updating array state
  const addHobby = (hobby) => {
    setHobbies([...hobbies, hobby]);
  };

  const removeHobby = (hobbyToRemove) => {
    setHobbies(hobbies.filter(h => h !== hobbyToRemove));
  };

  return (
    <div>
      {isEditing ? (
        <div>
          <input 
            value={name} 
            onChange={e => setName(e.target.value)} 
          />
          <button onClick={() => setIsEditing(false)}>Save</button>
        </div>
      ) : (
        <div>
          <h2>{name}</h2>
          <button onClick={() => setIsEditing(true)}>Edit</button>
        </div>
      )}

      <ul>
        {hobbies.map(hobby => (
          <li key={hobby}>
            {hobby}
            <button onClick={() => removeHobby(hobby)}>Remove</button>
          </li>
        ))}
      </ul>
      <button onClick={() => addHobby('Gaming')}>Add Hobby</button>
    </div>
  );
}
```

### Functional State Updates

When the new state depends on the previous state, use the functional update form:

```jsx
function Counter() {
  const [count, setCount] = useState(0);

  // Bad — relies on stale closure in async scenarios
  const incrementWrong = () => {
    setTimeout(() => {
      setCount(count + 1); // May use outdated count value
    }, 1000);
  };

  // Good — functional update ensures latest state
  const incrementCorrect = () => {
    setTimeout(() => {
      setCount(prevCount => prevCount + 1); // Always uses latest value
    }, 1000);
  };

  // Multiple rapid updates — functional form is essential
  const tripleIncrement = () => {
    setCount(c => c + 1);
    setCount(c => c + 1);
    setCount(c => c + 1);
  };

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={incrementCorrect}>Delayed +1</button>
      <button onClick={tripleIncrement}>+3</button>
    </div>
  );
}
```

### Lazy State Initialization

For expensive initial state calculations, use a function to avoid recomputation on every render:

```jsx
function ExpensiveComponent() {
  // Bad — runs parseExpensiveData() on every render (ignores result after first)
  const [data, setData] = useState(parseExpensiveData(rawData));

  // Good — only runs once on initial render
  const [data, setData] = useState(() => parseExpensiveData(rawData));

  // Good — lazy init with localStorage
  const [theme, setTheme] = useState(() => {
    return localStorage.getItem('theme') || 'light';
  });

  return <div>{/* ... */}</div>;
}
```

### State in Multiple Components

```jsx
// Lifting state up — share state between siblings via parent
function TemperatureConverter() {
  const [celsius, setCelsius] = useState('');

  const handleCelsiusChange = (value) => {
    setCelsius(value);
  };

  const handleFahrenheitChange = (value) => {
    setCelsius(value !== '' ? (parseFloat(value) - 32) * 5 / 9 : '');
  };

  const fahrenheit = celsius !== '' ? parseFloat(celsius) * 9 / 5 + 32 : '';

  return (
    <div>
      <TemperatureInput
        scale="c"
        value={celsius}
        onChange={handleCelsiusChange}
      />
      <TemperatureInput
        scale="f"
        value={fahrenheit}
        onChange={handleFahrenheitChange}
      />
    </div>
  );
}

function TemperatureInput({ scale, value, onChange }) {
  const label = scale === 'c' ? 'Celsius' : 'Fahrenheit';

  return (
    <div>
      <label>Temperature in {label}:</label>
      <input
        type="number"
        value={value}
        onChange={e => onChange(e.target.value)}
      />
    </div>
  );
}
```
