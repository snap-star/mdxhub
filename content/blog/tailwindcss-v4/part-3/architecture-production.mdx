```css

@variant group-hocus (.group:hover &, .group:focus-visible &);

@variant is-valid (&[data-valid="true"]);
@variant is-invalid (&[data-invalid="true"]);
@variant is-loading (&[aria-busy="true"]);
@variant is-empty (&:is(:empty, [data-empty="true"]));

/* Motion preference variants */
@variant motion-safe (@media (prefers-reduced-motion: no-preference));
@variant motion-reduce (@media (prefers-reduced-motion: reduce));

/* High contrast mode */
@variant high-contrast (@media (prefers-contrast: more));

/* Viewport orientation */
@variant landscape (@media (orientation: landscape));
@variant portrait (@media (orientation: portrait));

/* Pointer type */
@variant pointer-coarse (@media (pointer: coarse));
@variant pointer-fine (@media (pointer: fine));

@variant hoverable (@media (hover: hover) and (pointer: fine));
```

### 14.2 Focus Management

```tsx
// Consistent focus styles
<button className="focus-visible:outline-none focus-visible:ring-2 
                   focus-visible:ring-blue-500 focus-visible:ring-offset-2 
                   focus-visible:ring-offset-white">
  Accessible button

// Focus trap for modals
export function FocusTrap({ children }: { children: React.ReactNode }) {
  const containerRef = useRef<HTMLDivElement>(null);
  
  useEffect(() => {
    const container = containerRef.current;
    if (!container) return;
    
    const focusableElements = container.querySelectorAll(
      'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
    );
    const firstElement = focusableElements[0] as HTMLElement;
    const lastElement = focusableElements[focusableElements.length - 1] as HTMLElement;
    
    firstElement?.focus();
    
    const handleTabKey = (e: KeyboardEvent) => {
      if (e.key !== 'Tab') return;
      
      if (e.shiftKey && document.activeElement === firstElement) {
        e.preventDefault();
        lastElement?.focus();
      } else if (!e.shiftKey && document.activeElement === lastElement) {
        e.preventDefault();
        firstElement?.focus();
      }
    };
    
    container.addEventListener('keydown', handleTabKey);
    return () => container.removeEventListener('keydown', handleTabKey);
  }, []);
  
  return <div ref={containerRef}>{children}</div>;
}
```

```tsx
    const button = screen.getByRole('button', { name: /click me/i });
    
    // Test that critical classes are present
    expect(button.className).toContain('inline-flex');
    expect(button.className).toContain('rounded-lg');
    expect(button.className).toContain('font-medium');

    expect(button.className).toContain('hover:bg-red-700');
  });

    const button = screen.getByRole('button', { name: /custom/i });
    
    expect(button.className).toContain('my-custom-class');
    // Tailwind-merge should still have base classes
    expect(button.className).toContain('rounded-lg');
  });

    render(<Button disabled>Disabled</Button>);
    const button = screen.getByRole('button', { name: /disabled/i });
    
    expect(button).toBeDisabled();

  });
```
